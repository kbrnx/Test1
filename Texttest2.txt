    <h1>Deine zufällige Aufgabe:</h1>
    <div class="task-container" id="task-box"></div>
    
    <button class="button" onclick="generateTask()">Neue Aufgabe</button>

    <script>
        var tasks = [
            "Küsse deinen Partner an der sinnlichsten Stelle.",
            "Flüstere deinem Partner ins Ohr, was du gerade tun möchtest.",
            "Streiche sanft über die Haut deines Partners und sieh, wie er/sie reagiert."
        ];

        function generateTask() {
            var taskBox = document.getElementById("task-box");
            taskBox.innerText = tasks[Math.floor(Math.random() * tasks.length)];
            taskBox.classList.add("show");
        }

        // Dekorative Glühpunkte generieren
        function createGlowEffect() {
            for (let i = 0; i < 10; i++) {
                let glow = document.createElement("div");
                glow.classList.add("glow");
                glow.style.left = Math.random() * 100 + "vw";
                glow.style.top = Math.random() * 100 + "vh";
                glow.style.animationDelay = Math.random() * 2 + "s";
                document.body.appendChild(glow);
            }
        }

        createGlowEffect();
    </script>
